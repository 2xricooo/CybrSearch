async function searchData() { document.getElementById("results").innerHTML = ""; const email = document.getElementById("email").value.trim(); const phone = document.getElementById("phone").value.trim(); const username = document.getElementById("username").value.trim(); const discord = document.getElementById("discord").value.trim(); const ip = document.getElementById("ip").value.trim(); if (!email) updateResult("e", "⚠️ No email entered.", "warning"); else updateResult("e", `✅ Email checked (demo): ${email}`, "success"); if (!phone) updateResult("p", "ℹ️ Phone check skipped.", "warning"); else updateResult("p", `✅ Phone checked (demo): ${phone}`, "success"); if (!username) updateResult("u", "ℹ️ Username check skipped.", "warning"); else updateResult("u", `✅ Username checked (demo): ${username}`, "success"); if (!discord) updateResult("d", "ℹ️ Discord check skipped.", "warning"); else updateResult("d", `✅ Discord ID checked (demo): ${discord}`, "success"); if (ip) { // IPWHO.IS (geo + hostname) try { const geo = await fetch(`https://ipwho.is/${ip}`).then(res => res.json()); if (geo.success) { updateResult("geo", `🌐 Detected IP: ${geo.ip} Hostname: ${geo.connection?.domain || "N/A"} Country: ${geo.country} City: ${geo.city} Region: ${geo.region} Timezone: ${geo.timezone} ISP: ${geo.connection?.isp || "N/A"} Org: ${geo.org}`, "success"); } else { updateResult("geo", `⚠️ Geo lookup failed: ${geo.message}`, "warning"); } } catch { updateResult("geo", "❌ Error fetching Geo data.", "error"); } // ARIN RDAP Lookup try { const rdap = await fetch(`https://rdap.arin.net/registry/ip/${ip}`).then(r => r.json()); const handle = rdap.handle || "N/A"; const name = rdap.name || "N/A"; const startAddress = rdap.startAddress || "N/A"; const endAddress = rdap.endAddress || "N/A"; const orgName = rdap.entities?.[0]?.vcardArray?.[1]?.find(item => item[0] === "fn")?.[3] || "N/A"; const orgAddress = rdap.entities?.[0]?.vcardArray?.[1]?.find(item => item[0] === "adr")?.[3]?.join(", ") || "N/A"; updateResult("arin", `🌐 ARIN Info Handle: ${handle} Net Name: ${name} Start Address: ${startAddress} End Address: ${endAddress} Org Name: ${orgName} Org Address: ${orgAddress} More: https://search.arin.net/rdap/?query=${ip}`, "success"); } catch (err) { updateResult("arin", "❌ Error fetching ARIN RDAP info.", "error"); } } else { updateResult("ip", "ℹ️ IP geo-location and ARIN check skipped.", "warning"); } }
